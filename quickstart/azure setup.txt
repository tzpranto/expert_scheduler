# Create GPT partition table + single partition
sudo parted -s /dev/sdc mklabel gpt
sudo parted -s /dev/sdc mkpart primary ext4 0% 100%
# Format it (will create /dev/sdc1)

sudo mount /dev/sdc1 /mnt/ssd

//Verify

// record the uuid
sudo blkid /dev/sdc1

// Edit fstab
sudo nano /etc/fstab
// Append the following
